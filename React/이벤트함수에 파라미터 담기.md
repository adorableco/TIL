#이벤트 함수에 파라미터 담는 법

화살표 함수를 이용하여 원래 부르려던 함수를 감싸고, 해당 함수의 호출부에서 인자를 넘겨주면 된다.

```javascript
const clickHandler = (params, e) => {
  console.log(params); // error
  e.preventDefault();
  // do someting...
};

return (
  <button
    onClick={(e) => {
      clickHandler(params, e);
    }}
  >
    {" "}
    Do Something!
  </button>
);
```
